41  size_t cc = 0; // number of times run
42 
43  while(!terminate_) {
44  // wait until terminate or one more item
45  lock.lock();
46  {
47  // reduce count by number of times we have since ran func_()
48  count_ -= cc;
49  cv_.wait(lock, [&]() { return terminate_ || count_ > 0; });
50  cc = count_; // current count is the # of times to run func_() now
51  }
52  lock.unlock(); // relinquish lock
53 
54  // call function cc times in a row, outside of lock so notify() will not block
55  // for long-lasting functions
56  for (size_t i=0; i<cc; ++i) {
57  func_();
58  }
59  }
60  }
