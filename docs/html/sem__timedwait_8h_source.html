6 /*
7  * s e m _ t i m e d w a i t
8  *
9  * Function:
10  * Implements a version of sem_timedwait().
11  *
12  * Description:
13  * Not all systems implement sem_timedwait(), which is a version of
14  * sem_wait() with a timeout. Mac OS X is one example, at least up to
15  * and including version 10.6 (Leopard). If such a function is needed,
16  * this code provides a reasonable implementation, which I think is
17  * compatible with the standard version, although possibly less
18  * efficient. It works by creating a thread that interrupts a normal
19  * sem_wait() call after the specified timeout.
20  *
21  * Call:
22  *
23  * The Linux man pages say:
24  *
25  * #include <semaphore.h>
26  *
27  * int sem_timedwait(sem_t *sem, const struct timespec *abs_timeout);
28  *
29  * sem_timedwait() is the same as sem_wait(), except that abs_timeout
30  * specifies a limit on the amount of time that the call should block if
31  * the decrement cannot be immediately performed. The abs_timeout argument
32  * points to a structure that specifies an absolute timeout in seconds and
33  * nanoseconds since the Epoch (00:00:00, 1 January 1970). This structure
34  * is defined as follows:
35  *
36  * struct timespec {
37  * time_t tv_sec; Seconds
38  * long tv_nsec; Nanoseconds [0 .. 999999999]
39  * };
40  *
41  * If the timeout has already expired by the time of the call, and the
42  * semaphore could not be locked immediately, then sem_timedwait() fails
43  * with a timeout error (errno set to ETIMEDOUT).
44  * If the operation can be performed immediately, then sem_timedwait()
45  * never fails with a timeout error, regardless of the value of abs_timeout.
46  * Furthermore, the validity of abs_timeout is not checked in this case.
47  *
48  * Limitations:
49  *
50  * The mechanism used involves sending a SIGUSR2 signal to the thread
51  * calling sem_timedwait(). The handler for this signal is set to a null
52  * routine which does nothing, and with any flags for the signal
53  * (eg SA_RESTART) cleared. Note that this effective disabling of the
54  * SIGUSR2 signal is a side-effect of using this routine, and means it
55  * may not be a completely transparent plug-in replacement for a
56  * 'normal' sig_timedwait() call. Since OS X does not declare the
57  * sem_timedwait() call in its standard include files, the relevant
58  * declaration (shown above in the man pages extract) will probably have
59  * to be added to any code that uses this.
60  *
61  * Compiling:
62  * This compiles and runs cleanly on OS X (10.6) with gcc with the
63  * -Wall -ansi -pedantic flags. On Linux, using -ansi causes a sweep of
64  * compiler complaints about the timespec structure, but it compiles
65  * and works fine with just -Wall -pedantic. (Since Linux provides
66  * sem_timedwait() anyway, this really isn't needed on Linux.) However,
67  * since Linux provides sem_timedwait anyway, the sem_timedwait()
68  * code in this file is only compiled on OS X, and is a null on other
69  * systems.
70  *
71  * Testing:
72  * This file contains a test program that exercises the sem_timedwait
73  * code. It is compiled if the pre-processor variable TEST is defined.
74  * For more details, see the comments for the test routine at the end
75  * of the file.
76  *
77  * Author: Keith Shortridge, AAO.

182 inline int sem_timedwait (
183  sem_t *sem,
184  const struct timespec *abs_timeout)
185 {
186  int result = 0; /* Code returned by this routine 0 or -1 */
187 
188  /* "Under no circumstances shall the function fail if the semaphore
189  * can be locked immediately". So we try to get it quickly to see if we
190  * can avoid all the timeout overheads.
191  */
192 
193  if (sem_trywait(sem) == 0) {
194 
195  /* Yes, got it immediately. */
196 
197  result = 0;
198 
199  } else {
200 
201  /* No, we've got to do it with a sem_wait() call and a thread to run
202  * the timeout. First, work out the time from now to the specified
203  * timeout, which we will pass to the timeout thread in a way that can
204  * be used to pass to nanosleep(). So we need this in seconds and
205  * nanoseconds. Along the way, we check for an invalid passed time,
206  * and for one that's already expired.
207  */
208 
209  if ((abs_timeout->tv_nsec < 0) || (abs_timeout->tv_nsec > 1000000000)) {
210 
211  /* Passed time is invalid */
212 
213  result = -1;
214  errno = EINVAL;
215 
216  } else {
217 
218  struct timeval currentTime; /* Time now */
219  long secsToWait,nsecsToWait; /* Seconds and nsec to delay */
220  gettimeofday (&currentTime,NULL);
221  secsToWait = abs_timeout->tv_sec - currentTime.tv_sec;
222  nsecsToWait = (abs_timeout->tv_nsec - (currentTime.tv_usec * 1000));
223  while (nsecsToWait < 0) {
224  nsecsToWait += 1000000000;
225  secsToWait--;
226  }
227  if ((secsToWait < 0) || ((secsToWait == 0) && (nsecsToWait < 0))) {
228 
229  /* Time has passed. Report an immediate timeout. */
230 
231  result = -1;
232  errno = ETIMEDOUT;
233 
234  } else {
235 
236  /* We're going to have to do a sem_wait() with a timeout thread.
237  * The thread will wait the specified time, then will issue a
238  * SIGUSR2 signal that will interrupt the sem_wait() call.
239  * We pass the thread the id of the current thread, the delay,
240  * and the address of a flag to set on a timeout, so we can
241  * distinguish an interrupt caused by the timeout thread from
242  * one caused by some other signal.
243  */
244 
245  volatile short timedOut; /* Flag to set on timeout */
246  timeoutDetails details; /* All the stuff the thread must know */
247  struct sigaction oldSignalAction; /* Current signal setting */
248  pthread_t timeoutThread; /* Id of timeout thread */
249  cleanupDetails cleaningDetails; /* What the cleanup routine needs */
250  int oldCancelState; /* Previous cancellation state */
251  int ignoreCancelState; /* Used in call, but ignored */
252  int createStatus; /* Status of pthread_create() call */
253 
254  /* If the current thread is cancelled (and CML does do this)
255  * we don't want to leave our timer thread running - if we've
256  * started the thread we want to make sure we join it in order
257  * to release its resources. So we set a cleanup handler to
258  * do this. We pass it the address of the structure that will
259  * hold all it needs to know. While we set all this up,
260  * we prevent ourselves being cancelled, so all this data is
261  * coherent.
262  */
263 
264  pthread_setcancelstate (PTHREAD_CANCEL_DISABLE,&oldCancelState);
265  timeoutThread = (pthread_t) 0;
266  cleaningDetails.timedOutShort = &timedOut;
267  cleaningDetails.threadIdAddr = &timeoutThread;
268  cleaningDetails.sigHandlerAddr = &oldSignalAction;
269  pthread_cleanup_push (timeoutThreadCleanup,&cleaningDetails);
270 
271  /* Set up the details for the thread. Clear the timeout flag,
272  * record the current SIGUSR2 action settings so we can restore
273  * them later.
274  */
275 
276  details.delay.tv_sec = secsToWait;
277  details.delay.tv_nsec = nsecsToWait;
278  details.callingThread = pthread_self();
279  details.timedOutShort = &timedOut;
280  timedOut = FALSE;
281  sigaction (SIGUSR2,NULL,&oldSignalAction);
282 
283  /* Start up the timeout thread. Once we've done that, we can
284  * restore the previous cancellation state.
285  */
286 
287  createStatus = pthread_create(&timeoutThread,NULL,
288  timeoutThreadMain, (void*)&details);
289  pthread_setcancelstate (oldCancelState,&ignoreCancelState);
290 
291  if (createStatus < 0) {
292 
293  /* Failed to create thread. errno will already be set properly */
294 
295  result = -1;
296 
297  } else {
298 
299  /* Thread created OK. This is where we wait for the semaphore.
300  */
301 
302  if (sem_wait(sem) == 0) {
303 
304  /* Got the semaphore OK. We return zero, and all's well. */
305 
306  result = 0;
307 
308  } else {
309 
310  /* If we got a -1 error from sem_wait(), it may be because
311  * it was interrupted by a timeout, or failed for some
312  * other reason. We check for the expected timeout
313  * condition, which is an 'interrupted' status and the
314  * timeout flag set by the timeout thread. We report that as
315  * a timeout error. Anything else is some other error and
316  * errno is already set properly.
317  */
318 
319  result = -1;
320  if (errno == EINTR) {
321  if (timedOut) errno = ETIMEDOUT;
322  }
323  }
324 
325  }
326 
327  /* The cleanup routine - timeoutThreadCleanup() - packages up
328  * any tidying up that is needed, including joining with the
329  * timer thread. This will be called if the current thread is
330  * cancelled, but we need it to happen anyway, so we set the
331  * execute flag true here as we remove it from the list of
332  * cleanup routines to be called. So normally, this line amounts
333  * to calling timeoutThreadCleanup().
334  */
335 
336  pthread_cleanup_pop (TRUE);
337  }
338  }
339  }
340  return (result);
341 }

344 /*
345  * t i m e o u t T h r e a d C l e a n u p
346  *
347  * This internal routine tidies up at the end of a sem_timedwait() call.
348  * It is set as a cleanup routine for the current thread (not the timer
349  * thread) so it is executed even if the thread is cancelled. This is
350  * important, as we need to tidy up the timeout thread. If we took the
351  * semaphore (in other words, if we didn't timeout) then the timer thread
352  * will still be running, sitting in its nanosleep() call, and we need
353  * to cancel it. If the timer thread did signal a timeout then it will
354  * now be closing down. In either case, we need to join it (using a call
355  * to pthread_join()) or its resources will never be released.
356  * The single argument is a pointer to a cleanupDetails structure that has
357  * all the routine needs to know.
358  */
359 
360 inline void timeoutThreadCleanup (void* passedPtr)
361 {
362  /* Get what we need from the structure we've been passed. */
363 
364  cleanupDetails *detailsPtr = (cleanupDetails*) passedPtr;
365  short timedOut = *(detailsPtr->timedOutShort);
366  pthread_t timeoutThread = *(detailsPtr->threadIdAddr);
367 
368  /* If we created the thread, stop it - doesn't matter if it's no longer
369  * running, pthread_cancel can handle that. We make sure we wait for it
370  * to complete, because it is this pthread_join() call that releases any
371  * memory the thread may have allocated. Note that cancelling a thread is
372  * generally not a good idea, because of the difficulty of cleaning up
373  * after it, but this is a very simple thread that does nothing but call
374  * nanosleep(), and that we can cancel quite happily.
375  */
376 
377  if (!timedOut) pthread_cancel(timeoutThread);
378  pthread_join(timeoutThread,NULL);
379 
380  /* The code originally restored the old action handler, which generally
381  * was the default handler that caused the task to exit. Just occasionally,
382  * there seem to be cases where the signal is still queued and ready to
383  * trigger even though the thread that presumably sent it off just before
384  * it was cancelled has finished. I had thought that once we'd joined
385  * that thread, we could be sure of not seeing the signal, but that seems
386  * not to be the case, and so restoring a handler that will allow the task
387  * to crash is not a good idea, and so the line below has been commented
388  * out.
389  *
390  * sigaction (SIGUSR2,detailsPtr->sigHandlerAddr,NULL);
391  */
392 }

394 /* -------------------------------------------------------------------------- */
395 /*
396  * t i m e o u t T h r e a d M a i n
397  *
398  * This internal routine is the main code for the timeout thread.
399  * The single argument is a pointer to a timeoutDetails structure that has
400  * all the thread needs to know - thread to signal, delay time, and the
401  * address of a flag to set if it triggers a timeout.
402  */
403 
404 inline void* timeoutThreadMain (void* passedPtr)
405 {
406  void* Return = (void*) 0;
407 
408  /* We grab all the data held in the calling thread right now. In some
409  * cases, we find that the calling thread has vanished and released
410  * its memory, including the details structure, by the time the timeout
411  * expires, and then we get an access violation when we try to set the
412  * 'timed out' flag.
413  */
414 
415  timeoutDetails details = *((timeoutDetails*) passedPtr);
416  struct timespec requestedDelay = details.delay;
417 
418  /* We do a nanosleep() for the specified delay, and then trigger a
419  * timeout. Note that we allow for the case where the nanosleep() is
420  * interrupted, and restart it for the remaining time. If the
421  * thread that is doing the sem_wait() call gets the semaphore, it
422  * will cancel this thread, which is fine as we aren't doing anything
423  * other than a sleep and a signal.
424  */
425 
426  for (;;) {
427  struct timespec remainingDelay;
428  if (nanosleep (&requestedDelay,&remainingDelay) == 0) {
429  break;
430  } else if (errno == EINTR) {
431  requestedDelay = remainingDelay;
432  } else {
433  Return = (void*) (uintptr_t)errno;
434  break;
435  }
436  }
437 
438  /* We've completed the delay without being cancelled, so we now trigger
439  * the timeout by sending a signal to the calling thread. And that's it,
440  * although we set the timeout flag first to indicate that it was us
441  * that interrupted the sem_wait() call. One precaution: before we
442  * try to set the timed-out flag, make sure the calling thread still
443  * exists - this may not be the case if things are closing down a bit
444  * messily. We check this quickly using a zero test signal.
445  */
446 
447  if (pthread_kill(details.callingThread,0) == 0) {
448  *(details.timedOutShort) = TRUE;
449  if (triggerSignal (SIGUSR2,details.callingThread) < 0) {
450  Return = (void*) (uintptr_t)errno;
451  }
452  }
453 
454  return Return;
455 }

461  * This is a general purpose routine that sends a specified signal to
462  * a specified thread, setting up a signal handler that does nothing,

468 inline int triggerSignal (int Signal, pthread_t Thread)
469 {
470  int Result = 0;
471  struct sigaction SignalDetails;
472  SignalDetails.sa_handler = ignoreSignal;
473  SignalDetails.sa_flags = 0;
474  (void) sigemptyset(&SignalDetails.sa_mask);
475  if ((Result = sigaction(Signal,&SignalDetails,NULL)) == 0) {
476  Result = pthread_kill(Thread,Signal);
477  }
478  return Result;
479 }
