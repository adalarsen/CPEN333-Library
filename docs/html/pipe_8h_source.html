137  bool write(const void* data, size_t size) {
138 
139  // check we are opened in write mode
140  if (mode_ != mode::WRITE) {
141  return false;
142  }
143 
144  uint8_t *ptr = (uint8_t *) data;
145 
146  // try to write bytes
147  std::unique_lock<decltype(wmutex_)> lock(wmutex_, std::defer_lock);
148  for (size_t i = 0; i < size; ++i) {
149  consumer_.wait(); // wait until there is space in the pipe
150 
151  // write next byte and advance write index
152  lock.lock();
153  int pos = info_->write;
154 
155  // check for EOF
156  if (info_->weof > 0) {
157  // Do not start write if pipe is closed, otherwise finish writing
158  if ( (i == 0) // start
159  || (pos == 0 && info_->weof == info_->size) // wrap
160  || (pos == info_->weof)) { // regular
161  consumer_.notify(); // notify any other writer threads
162  return false;
163  }
164  }
165 
166  // next byte location to write, wrapping around if need be
167  if ((++(info_->write)) == info_->size) {
168  info_->write = 0;
169  }
170  lock.unlock();
171 
172  // do actual write outside of lock for efficiency
173  // (though here it is only one byte...)
174  pipe_[pos] = *ptr;
175  ++ptr;
176 
177  producer_.notify(); // byte available for read
178  }
179
