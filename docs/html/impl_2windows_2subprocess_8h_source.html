204  static void append_argument(const std::string &arg, std::string &commandline) {
205  if (!arg.empty() && arg.find_first_of(" \t\n\v\"") == arg.npos) {
206  commandline.append(arg);
207  } else {
208  // quote
209  commandline.push_back('"');
210  for (auto it = arg.begin(); it != arg.end(); ++it) {
211  unsigned num_slashes = 0;
212 
213  while (it != arg.end() && *it == '\\') {
214  ++it;
215  ++num_slashes;
216  }
217 
218  if (it == arg.end()) {
219  //
220  // Escape all backslashes, but let the terminating
221  // double quotation mark we add below be interpreted
222  // as a metacharacter.
223  //
224  commandline.append(num_slashes * 2, '\\');
225  break;
226  } else if (*it == '"') {
227  //
228  // Escape all backslashes and the following
229  // double quotation mark.
230  //
231  commandline.append(num_slashes * 2 + 1, '\\');
232  commandline.push_back(*it);
233  } else {
234  //
235  // Backslashes aren't special here.
236  //
237  commandline.append(num_slashes, '\\');
238  commandline.push_back(*it);
239  }
240  }

242  }
